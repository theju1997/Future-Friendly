Solution:-

        x=open('sample.txt','r+')
        x.write('Self Introduction \n' 'Thank you for giving me this opportunity to introduce my \n'  'My self Thejaswini TR\n' 'I am learning python Fullstack\n')
        print(x)
        x.close()
        char_count = 0
        word_count = 0
        line_count = 0
        with open('sample.txt', 'r') as file:
            for i in file:
                line_count += 1  # Increment the line count
                char_count += len(i)  # Increment the character count
                words = i.split()  # Split the line into words
                word_count += len(words)  # Increment the word count
        
        # Output the results
        print("Character count:", char_count)
        print("Word count:", word_count)
        print("Line count:", line_count)
