        with open('student.txt','r+')as f:
            text=f.read()
            print(text)
            print('the current cursor position',f.tell())
            f.seek(17)
            print('the current cursor position',f.tell())
            f.write('Gems!!')
            print('the current cursor position',f.tell())
            f.seek(0)
